    <script>
        // Initialize app
        let students = JSON.parse(localStorage.getItem('students')) || [];
        let attendanceRecords = JSON.parse(localStorage.getItem('attendanceRecords')) || [];

        // Set today's date
        document.getElementById('attendanceDate').valueAsDate = new Date();

        function showTab(tabName) {
            // Hide all tabs
            document.querySelectorAll('.tab-content').forEach(tab => {
                tab.classList.remove('active');
            });
            document.querySelectorAll('.nav-tab').forEach(tab => {
                tab.classList.remove('active');
            });

            // Show selected tab
            document.getElementById(tabName).classList.add('active');
            event.target.classList.add('active');
            
            if (tabName === 'manage') renderStudents();
            if (tabName === 'reports') loadReports();
            if (tabName === 'attendance') loadCoursesForAttendance();
        }

        function addStudent() {
            const name = document.getElementById('studentName').value.trim();
            const id = document.getElementById('studentId').value.trim();
            const course = document.getElementById('course').value;

            if (!name || !id || !course) {
                alert('Please fill all fields');
                return;
            }

            if (students.find(s => s.id === id)) {
                alert('Student ID already exists');
                return;
            }

            const student = { id, name, course, attendance: {} };
            students.push(student);
            localStorage.setItem('students', JSON.stringify(students));

            document.getElementById('studentName').value = '';
            document.getElementById('studentId').value = '';
            document.getElementById('course').value = '';

            renderStudents();
            alert('Student added successfully!');
        }

        function renderStudents() {
            const container = document.getElementById('studentsList');
            container.innerHTML = '';

            students.forEach((student, index) => {
                const card = document.createElement('div');
                card.className = 'student-card';
                card.innerHTML = `
                    <div class="student-name">${student.name}</div>
                    <div class="student-id">ID: ${student.id}</div>
                    <div style="color: #666; margin-bottom: 15px;">Course: ${student.course}</div>
                    <div style="display: flex; gap: 10px;">
                        <button class="btn btn-danger" onclick="deleteStudent(${index})" style="padding: 10px;">Delete</button>
                    </div>
                `;
                container.appendChild(card);
            });
        }

        function deleteStudent(index) {
            if (confirm('Are you sure you want to delete this student?')) {
                students.splice(index, 1);
                localStorage.setItem('students', JSON.stringify(students));
                renderStudents();
            }
        }

        function loadCoursesForAttendance() {
            const courseSelect = document.getElementById('attendanceCourse');
            courseSelect.innerHTML = '<option value="">Select Course</option>';
            
            const courses = [...new Set(students.map(s => s.course))];
            courses.forEach(course => {
                const option = document.createElement('option');
                option.value = course;
                option.textContent = course;
                courseSelect.appendChild(option);
            });
        }

        function loadStudentsForAttendance() {
            const course = document.getElementById('attendanceCourse').value;
            const container = document.getElementById('attendanceStudents');
            container.innerHTML = '';

            if (!course) return;

            const courseStudents = students.filter(s => s.course === course);
            
            courseStudents.forEach(student => {
                const date = document.getElementById('attendanceDate').value;
                const status = student.attendance[date] || 'absent';
                
                const card = document.createElement('div');
                card.className = 'student-card';
                card.innerHTML = `
                    <div class="student-name">${student.name}</div>
                    <div class="student-id">ID: ${student.id}</div>
                    <div class="attendance-buttons">
                        <button class="attendance-btn btn-present" onclick="markStudent('${student.id}', 'present')" ${status === 'present' ? 'disabled' : ''}>Present</button>
                        <button class="attendance-btn btn-absent" onclick="markStudent('${student.id}', 'absent')" ${status === 'absent' ? 'disabled' : ''}>Absent</button>
                    </div>
                `;
                container.appendChild(card);
            });
        }

        function markStudent(studentId, status) {
            const date = document.getElementById('attendanceDate').value;
            const student = students.find(s => s.id === studentId);
            
            if (student) {
                student.attendance[date] = status;
                localStorage.setItem('students', JSON.stringify(students));
                loadStudentsForAttendance();
            }
        }

        function markAttendance() {
            const date = document.getElementById('attendanceDate').value;
            const course = document.getElementById('attendanceCourse').value;
            
            if (!date || !course) {
                alert('Please select date and course');
                return;
            }

            // Save to records
            students.filter(s => s.course === course).forEach(student => {
                if (student.attendance[date]) {
                    attendanceRecords.push({
                        date,
                        studentId: student.id,
                        studentName: student.name,
                        course,
                        status: student.attendance[date]
                    });
                }
            });

            localStorage.setItem('attendanceRecords', JSON.stringify(attendanceRecords));
            localStorage.setItem('students', JSON.stringify(students));
            
            alert('Attendance saved successfully!');
            loadStudentsForAttendance();
        }

        function loadReports() {
            // Update stats
            document.getElementById('totalStudents').textContent = students.length;
            document.getElementById('totalClasses').textContent = new Set(attendanceRecords.map(r => r.date)).size;
            
            const totalAttendance = attendanceRecords.filter(r => r.status === 'present').length;
            const totalRecords = attendanceRecords.length;
            const avg = totalRecords ? Math.round((totalAttendance / totalRecords) * 100) : 0;
            document.getElementById('avgAttendance').textContent = avg + '%';

            // Update table
            const tbody = document.getElementById('reportsBody');
            tbody.innerHTML = '';

            attendanceRecords.slice(-50).reverse().forEach(record => { // Last 50 records
                const row = document.createElement('tr');
                row.innerHTML = `
                    <td>${new Date(record.date).toLocaleDateString()}</td>
                    <td>${record.studentName} (${record.studentId})</td>
                    <td>${record.course}</td>
                    <td class="status-${record.status}">${record.status.toUpperCase()}</td>
                `;
                tbody.appendChild(row);
            });
        }

        // Initial load
        renderStudents();
        loadCoursesForAttendance();
    </script>
